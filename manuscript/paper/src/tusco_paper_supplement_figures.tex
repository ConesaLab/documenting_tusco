	  \section{Supplementary Figures}
		  This section compiles visual summaries referenced throughout the manuscript.
		  Supplementary Fig.~1 presents parameter selection sensitivity analyses for the TUSCO benchmarking framework.
		  Supplementary Fig.~2 contrasts fundamental transcript properties between
		  the TUSCO gene set and canonical annotations in human and mouse, while
		  Supplementary Fig.~3 benchmarks the expression and splicing invariance of
		  genes in the TUSCO gene set against other single-isoform loci. Supplementary Fig.~4 shows false-negative
		  rates from TUSCO evaluation grouped by sequencing platform and method. Supplementary Fig.~5 compares
		  sensitivity, false-negative rate, and false discovery rate between TUSCO
		  evaluation and MAJIQ-L across all pipelines. Supplementary Fig.~6 examines
		  per-read coverage distributions for transcripts in the TUSCO gene set, comparing TP and PTP calls and stratified by reference transcript exonic length.
		  Supplementary Fig.~7 evaluates how SIRV-based concordance metrics respond to replicate
		  number, contextualizing the stability of spike-in controls relative to
		  endogenous benchmarks.

  \begin{figure}[htbp]
      \centering
      \includegraphics[width=\textwidth]{../assets/fig/fig-s7.pdf}
      \caption{\textbf{Parameter selection sensitivity analysis for the TUSCO benchmarking framework.} \textbf{a}, TSS/CAGE window size validation showing cumulative fraction of genes whose annotated TSS (from refTSS database) falls within varying window sizes; ``TUSCO'' curves show within-gene sensitivity, while ``GENCODE'' curves show inter-gene collision rates. At $\pm$300~bp, human data show 96.6\% sensitivity against only 2.1\% inter-gene collision. \textbf{b}, Distribution of refTSS peak counts per gene at varying window sizes; values above 1.0 indicate potential ambiguity from multiple TSS peaks. \textbf{c}, Expression threshold sweep comparing TUSCO false-negative rates (from LRGASP evaluation) against SIRV baselines (dashed line) across four threshold configurations (Strict to More Lenient) for human and mouse. The ``Current'' configuration balances gene count against a sharp rise in false negatives under more lenient settings. \textbf{d}, Distribution of AlphaGenome-predicted tissue-median RPKM for genes in the TUSCO gene set (colored) versus other single-isoform genes (gray); dashed lines indicate medians for genes in the TUSCO gene set, demonstrating that they occupy the high-expression tail (90th--97th percentile). \textbf{e}, Novel splice junction threshold validation using AlphaGenome-predicted junction read counts, showing ECDF of the ratio $\max(\text{novel junction reads})/\mu_{\text{annotated}}$ for genes in the TUSCO gene set versus other single-isoform genes. Genes in the TUSCO gene set cluster well below the chosen $\alpha$ threshold (dashed lines; 0.01 for human, 0.05 for mouse), whereas excluded genes show markedly higher ratios.}
	      \label{fig:s1}

      \caption{\textbf{Genes in the TUSCO gene set are highly expressed and display minimal novel splicing signal.} Violin--box plots compare AlphaGenome-predicted scores for genes in the TUSCO gene set with single-isoform genes annotated in both GENCODE and RefSeq for human and mouse. \textbf{a}, Expression score (gene-level tissue-median RPKM). \textbf{b}, Splicing score (for each gene, the maximum AlphaGenome-predicted unannotated-to-annotated splice-junction count ratio across tissues). Values are plotted as $\log_{10}$ (score $+ 1 \times 10^{-4}$). Across both species, genes in the TUSCO gene set exhibit substantially higher expression and lower novel splicing score ratios than other single-isoform genes, consistent with their selection as widely expressed, splicing-invariant genes. Box centers mark medians; boxes span the interquartile range; whiskers extend $1.5 \times$ IQR. Colors: TUSCO (Human) light green, TUSCO (Mouse) dark green, Other (Human) light gray, Other (Mouse) dark gray.}

  \begin{figure}[htbp]
      \centering
      \includegraphics[width=\textwidth]{../assets/fig/fig-s6.pdf}
      \caption{\textbf{SIRV spike-in controls exhibit minimal dependence on replicate number.} Concordance metrics (sensitivity, non-redundant precision, inverse redundancy, 1-FDR, PDR, F1) were computed on FLAIR-identified SIRV transcripts from PacBio Sequel IIe cDNA libraries ($\sim$5 M reads per library). Metrics were summarized across all replicate combinations for 1--5 biological replicates (bars: mean with 95\% CI; points: individual combinations). Performance was essentially invariant with replicate number: precision, inverse redundancy and 1-FDR were $\sim$100\% at all levels; brain sensitivity was 86.1\% (83.7--88.5) at 1 replicate and $\sim$85.5\% at 3--5; kidney sensitivity ranged 80.3\% (78.4--82.1; 2 replicates) to 87.0\% (5 replicates). F1 remained stable (brain $\sim$92.2\%; kidney $\sim$89.0--93.0\%). These results indicate that SIRV spike-ins yield highly consistent concordance irrespective of replicate count.}
	      \label{fig:s7}
	  \end{figure}
